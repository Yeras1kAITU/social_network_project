Try these API endpoints with tools like Postman or curl 

# Create a new post
curl -X POST http://localhost:3000/api/posts \
  -H "Content-Type: application/json" \
  -d '{"title":"New Study Material","content":"Content here","author":"Test User"}'

# Update post #1
curl -X PUT http://localhost:3000/api/posts/1 \
  -H "Content-Type: application/json" \
  -d '{"likes":50}'
